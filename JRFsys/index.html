import js, json from pyodide.ffi import create_proxy async def process(event): data = event.detail.to_py() coords = json.loads(data['geo']) tid = data['tid'] try: db = js.window.gtfsDB st_data = db.stopTimes.to_py() if tid not in st_data: return st_list = sorted(st_data[tid], key=lambda x: int(x['stop_sequence'])) stops_dict = db.stops.to_py() timeline = [] for r in st_list: sid = r['stop_id'] if sid in stops_dict: s = stops_dict[sid] timeline.append({ "name": s['stop_name'], "code": s['stop_code'], "arr": r['arrival_time'], "dep": r['departure_time'], "lat": float(s['stop_lat']), "lon": float(s['stop_lon']) }) js.window.updateUI(json.dumps(coords), json.dumps(timeline)) except Exception as e: js.console.error(str(e)) js.document.getElementById('loader').style.display = 'none' js.window.addEventListener("py-process", create_proxy(process))
