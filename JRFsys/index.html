import js, json, math, re from pyodide.ffi import create_proxy async def start_analysis(event): data = event.detail.to_py() tid = data['tid'] csv_content = data['csv'] try: db = js.window.gtfsDB trip = next((t for t in db.trips.to_py() if t['trip_id'] == tid), None) sid = trip.get('shape_id') # 1. Timeline st = sorted([s for s in db.stop_times.to_py() if s['trip_id'] == tid], key=lambda x: int(x['stop_sequence'])) timeline = [] for r in st: s_info = next((s for s in db.stops.to_py() if s['stop_id'] == r['stop_id']), None) timeline.append({"name": s_info['stop_name'], "arr": r['arrival_time'], "dep": r['departure_time'], "lat": float(s_info['stop_lat']), "lon": float(s_info['stop_lon'])}) # 2. Parse CSV (Passed from JS) lines = csv_content.lstrip('\ufeff').replace('\r\n', '\n').replace('\r', '\n').strip().split('\n') header = [re.sub(r'[^a-z0-9_]', '', x.lower().strip()) for x in lines[0].split(',')] idx = { 'id': header.index('shape_id'), 'lat': header.index('shape_pt_lat'), 'lon': header.index('shape_pt_lon'), 'seq': header.index('shape_pt_sequence') } geo = [] for l in lines[1:]: c = [col.strip() for col in l.split(',')] if len(c) > idx['id'] and c[idx['id']] == sid: geo.append([float(c[idx['lat']]), float(c[idx['lon']]), int(c[idx['seq']])]) geo.sort(key=lambda x: x[2]) coords = [[p[0], p[1]] for p in geo] # 3. Distance d = 0 for i in range(len(coords)-1): p1, p2 = coords[i], coords[i+1] a = math.sin(math.radians(p2[0]-p1[0])/2)**2 + math.cos(math.radians(p1[0]))*math.cos(math.radians(p2[0]))*math.sin(math.radians(p2[1]-p1[1])/2)**2 d += 6371 * 2 * math.atan2(math.sqrt(a), math.sqrt(1-a)) js.window.updateUI(json.dumps(coords), json.dumps(timeline), json.dumps({"id": tid, "dist": d})) except Exception as e: js.log(f"PY ERROR: {str(e)}", True) js.document.getElementById('loader').style.display = 'none' js.window.addEventListener("py-process", create_proxy(start_analysis)) js.log("Analyzer Engine v6.2 Standby.")
